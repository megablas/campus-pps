    <script>
        function toggleFaqSmooth(id) {
            const content = document.getElementById('content-' + id);
            const iconDiv = document.getElementById('icon-' + id);
            if (!content || !iconDiv) return;
            const iconSvg = iconDiv.querySelector('svg');
            if (!content.style.maxHeight || content.style.maxHeight === '0px') {
                // Cerrar otros
                document.querySelectorAll('[id^="content-faq-"]').forEach(el => {
                    el.style.maxHeight = '0px';
                    try {
                        const otherIcon = document.getElementById(el.id.replace('content-', 'icon-'));
                        otherIcon.style.backgroundColor = '#f1f5f9';
                        otherIcon.style.transform = 'rotate(0deg)';
                        otherIcon.querySelector('svg').style.stroke = '#64748b';
                    } catch (e) {}
                });
                // Abrir este
                content.style.maxHeight = content.scrollHeight + 50 + "px";
                iconDiv.style.backgroundColor = '#4B8C7A';
                iconDiv.style.transform = 'rotate(180deg)';
                iconSvg.style.stroke = '#ffffff';
            } else {
                content.style.maxHeight = '0px';
                iconDiv.style.backgroundColor = '#f1f5f9';
                iconDiv.style.transform = 'rotate(0deg)';
                iconSvg.style.stroke = '#64748b';
            }
        }
